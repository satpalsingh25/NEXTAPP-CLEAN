Create Audit Log system (lightweight, scalable, no performance impact)

----------------------------------------
STEP 1 — SCHEMA
----------------------------------------

Add new model:

model AuditLog {
  id           String   @id @default(uuid())
  company_id   String
  user_id      String?
  action       String   // CREATE_FOLDER, UPLOAD_FILE, LOGIN, etc.
  module       String   // DMS, COMPLIANCE, AMC, ADMIN
  entity_type  String?  // file, folder, user, etc.
  entity_id    String?
  description  String?
  created_at   DateTime @default(now())
}

----------------------------------------
RUN
----------------------------------------

npx prisma migrate dev

----------------------------------------
STEP 2 — LOG UTILITY
----------------------------------------

Create:

/lib/audit-log.ts

----------------------------------------
CODE
----------------------------------------

import prisma from "@/lib/prisma"

export async function logAudit({
  company_id,
  user_id,
  action,
  module,
  entity_type,
  entity_id,
  description
}) {
  try {
    await prisma.auditLog.create({
      data: {
        company_id,
        user_id,
        action,
        module,
        entity_type,
        entity_id,
        description
      }
    })
  } catch (err) {
    console.error("Audit log failed", err)
  }
}
