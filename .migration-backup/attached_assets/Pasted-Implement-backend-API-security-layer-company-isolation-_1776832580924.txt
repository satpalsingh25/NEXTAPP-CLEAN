Implement backend API security layer (company isolation + permission enforcement)

----------------------------------------
STEP 1 — CREATE PERMISSION HELPER
----------------------------------------

Create:

/lib/permission.ts

----------------------------------------
CODE
----------------------------------------

export function requireAuth(req) {
  if (!req.user) {
    throw new Error("Unauthorized")
  }
}

export function requireRole(req, roles = []) {
  if (!roles.includes(req.user.role)) {
    throw new Error("Forbidden")
  }
}

export function requireCompanyAccess(req, resourceCompanyId) {
  if (req.user.company_id !== resourceCompanyId) {
    throw new Error("Access denied")
  }
}
