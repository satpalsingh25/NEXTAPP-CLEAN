Implement Module Assignment modal for Super Admin.

----------------------------------------
STEP 1 — API (GET COMPANY MODULES)
----------------------------------------

GET /api/admin/company-modules?company_id=XXX

----------------------------------------
LOGIC
----------------------------------------

const modules = await prisma.module.findMany({
  include: {
    company_modules: {
      where: { company_id }
    }
  }
})

return modules.map(m => ({
  id: m.id,
  name: m.name,
  enabled: m.company_modules[0]?.enabled ?? true
}))

----------------------------------------
STEP 2 — API (UPDATE MODULES)
----------------------------------------

POST /api/admin/company-modules

Body:
{
  company_id,
  modules: [
    { module_id, enabled }
  ]
}

----------------------------------------
LOGIC
----------------------------------------

for (const m of modules) {
  await prisma.companyModule.upsert({
    where: {
      company_id_module_id: {
        company_id,
        module_id: m.module_id
      }
    },
    update: { enabled: m.enabled },
    create: {
      company_id,
      module_id: m.module_id,
      enabled: m.enabled
    }
  })
}

return { success: true }
